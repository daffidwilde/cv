\section{PhD thesis (ongoing)}
\cvitem{Title}{%
    \textbf{%
        New methods for algorithm evaluation and cluster initialisation with
        applications to healthcare%
    }
}
\cvitem{Supervisors}{\small%%
    Dr Jonathan Gillard, Dr Vincent Knight, Mr Kendal Smith (NHS Wales)
}
\cvitem{Description}{\small%
    The purpose of this project is to investigate and model variation in the
    service and cost of treating hospital patients in South Wales; this is
    achieved by identifying structures in routinely gathered, administrative
    healthcare datasets via clustering. In addition, this thesis considers some
    of the issues surrounding the process of evaluating an algorithm's quality,
    and presents novel methods for enriching the picture painted by traditional
    processes.
}
